\begin{center}
{\bf Uniqueness of the invariant measure}
\end{center}

\section*{Problem statement and interpretation}

Let \(E=C_0([0,1])\), \(H=L^2(0,1)\), and let \(S_t\) have kernel
\(p_t(x,y)\), the Dirichlet heat kernel for \(\frac12\Delta\).  I use the
Dirichlet extension of the Athreya--Butkovsky--L\^e--Mytnik (ABLM) mild
solution stipulated in the problem in the following explicit form: for
\(b_n=G_{1/n}\),
\[
        P_t^n\Phi(x)\longrightarrow P_t\Phi(x),
        \qquad x\in E,\quad t>0,\quad \Phi\in C_b(E).        \tag{A}
\]
The proof also needs the \(V(3/4)\) estimate for the drift part.  Since ABLM
state this only away from Dirichlet boundary conditions, I record the
Dirichlet replacement.  If
\[
        u_t=S_tx+V_t+K_t,\qquad
        V_t=\int_0^t S_{t-s}\,dW_s ,
\]
then, for every \(T<\infty\) and \(m\ge2\),
\[
 \sup_{0\le s<t\le T}\sup_{r\in[0,1]}
 \frac{\|K_t(r)-S_{t-s}K_s(r)\|_{L^m(\Omega)}}{|t-s|^{3/4}}<\infty . \tag{B}
\]

Here is the needed verification.  Put \(d(y)=y\wedge(1-y)\) and
\[
        \rho_\tau(y):=\int_0^\tau p_{2a}(y,y)\,da
        =\Var\{V_{s+\tau}(y)-S_\tau V_s(y)\}.
\]
The reflection formula for \(p_t\) gives, uniformly for \(0<\tau\le T\),
\[
        c(\sqrt{\tau}\wedge d(y))\le \rho_\tau(y)
        \le C(\sqrt{\tau}\wedge d(y)).                     \tag{1}
\]
Consequently, by splitting the integral into \(d(y)\le\sqrt{\tau}\) and its
complement and scaling the half-line kernel near each boundary,
\[
\sup_x\int_0^1p_a(x,y)\rho_\tau(y)^{-1/2}dy
       \le C(a^{-1/4}+\tau^{-1/4}),                         \tag{2}
\]
and
\[
\sup_x\int_0^1p_a(x,y)\rho_\tau(y)^{-1}dy
       \le C(a^{-1/2}+\tau^{-1/2}).                          \tag{3}
\]
Indeed, for example,
\(\int p_a(x,y)d(y)^{-1/2}dy\le Ca^{-1/4}\) and
\(\int p_a(x,y)d(y)^{-1}dy\le Ca^{-1/2}\); the apparent singularity at
\(y=0,1\) is cancelled by the Dirichlet factor \(p_a(x,y)=O(d(y))\).

Let \(b_n\ge0\) be smooth with \(\int b_n=1\).  Conditional on
\({\cal F}_s\), the random variable
\(V_r(y)-S_{r-s}V_s(y)\) is centered Gaussian with variance
\(\rho_{r-s}(y)\).  Hence
\[
 \E_s b_n(V_r(y)+S_{r-s}\kappa(y))
 \le (2\pi\rho_{r-s}(y))^{-1/2}                              \tag{4}
\]
for every \({\cal F}_s\)-measurable shift \(\kappa\).  Expanding integer
moments, ordering the times, and conditioning successively gives, from (2),
\[
 \left\|\int_s^t\!\!\int_0^1 p_{T-r}(x,y)
 b_n(V_r(y)+S_{r-s}\kappa(y))\,dy\,dr\right\|_{L^m}
      \le C_m(t-s)^{3/4}.                                    \tag{5}
\]
For the sewing increment, set
\[
 A_{s,t}^{T,n}(x)=\int_s^t\!\!\int_0^1p_{T-r}(x,y)
 b_n(V_r(y)+S_{r-s}\psi_s(y))\,dy\,dr,\qquad
 \delta A_{s,u,t}=A_{s,t}-A_{s,u}-A_{u,t}.
\]
The same conditioning argument with one derivative of the one-dimensional
heat semigroup uses (3) and gives
\[
 |\,\E_u\delta A_{s,u,t}^{T,n}(x)\,|
      \le C\,\lambda^T_{s,u}(x)(t-u)^{1/2},                   \tag{6}
\]
where \(\psi=S_\cdot x+K\) and
\(\lambda^T_{s,u}(x)=S_{T-u}K_u(x)-S_{T-s}K_s(x)\ge0\) is the random
control coming from positivity of the drift.  Estimates (5) and (6) are
precisely the two analytic inputs in the proof of ABLM Proposition 3.8; the
rest of that proof is the stochastic sewing lemma with random controls and
positivity.  Thus it carries over to the Dirichlet kernel and yields (B).

We prove that \(P_t\) has a unique invariant probability measure.

\section*{The reversible Gibbs measure}

Let \(\gamma\) be Brownian bridge measure on \(E\), i.e. the centered Gaussian
measure with covariance \((-\Delta_D)^{-1}\).  Define
\[
  \Lambda(v)=\int_0^1\1_{\{v(r)>0\}}\,dr,\qquad
  \pi(dv)=Z^{-1}e^{2\Lambda(v)}\,\gamma(dv).                 \tag{7}
\]
Then \(\pi\sim\gamma\) and has full support.  For \(B_n'=b_n\), with
\(B_n(a)=\int_{-\infty}^a b_n(r)\,dr\), put
\[
   \pi_n(dv)=Z_n^{-1}\exp\left(2\int_0^1B_n(v(r))\,dr\right)\gamma(dv).
\]
Since \(0\le B_n\le1\), \(B_n(a)\to\1_{\{a>0\}}\) for \(a\ne0\), and Brownian
bridge paths spend zero Lebesgue time at zero, dominated convergence gives
\[
        \|\pi_n-\pi\|_{\TV}\to0 .                            \tag{8}
\]

For fixed \(n\), the mild SPDE with drift \(b_n\) is the standard variational
gradient diffusion associated with the closed form
\[
        {\cal E}_n(F,G)=\frac12\int_E\langle DF,DG\rangle_H\,d\pi_n .
\]
Equivalently, on smooth cylinder functions,
\[
 L_nF(v)=\frac12{\rm Tr}\,D^2F(v)
       +\left\langle \frac12\Delta_Dv+b_n(v),DF(v)\right\rangle_H
\]
and Gaussian integration by parts gives
\[
 \int_E G\,L_nF\,d\pi_n
 =-\frac12\int_E\langle DF,DG\rangle_H\,d\pi_n
 =\int_E F\,L_nG\,d\pi_n .                                  \tag{9}
\]
Thus \(P_t^n\) is \(\pi_n\)-invariant and reversible.

\begin{proposition}\label{prop:gap}
The measure \(\pi\) is \(P_t\)-invariant.  Moreover \(P_t\) is symmetric on
\(L^2(\pi)\), and there exists \(c>0\) such that
\[
       \Var_\pi(P_t f)\le e^{-2ct}\Var_\pi(f),
       \qquad f\in L^2(\pi),\ t\ge0 .                       \tag{10}
\]
\end{proposition}

\begin{proof}
For \(f,g\in C_b(E)\), (A), (8), and the invariance and symmetry of
\((P_t^n,\pi_n)\) imply the following limits.  Indeed, if \(h_n\) are
uniformly bounded and \(h_n\to h\) pointwise, then
\(\int h_n\,d\pi_n\to\int h\,d\pi\) by total variation convergence plus
dominated convergence under the fixed measure \(\pi\).  Hence
\[
 \int P_tf\,d\pi=\lim_n\int P_t^nf\,d\pi_n=\int f\,d\pi
\]
and
\[
 \int gP_tf\,d\pi=\lim_n\int gP_t^nf\,d\pi_n
       =\lim_n\int fP_t^ng\,d\pi_n=\int fP_tg\,d\pi .
\]
This gives invariance and symmetry, first on \(C_b(E)\), then on \(L^2(\pi)\)
by density and Jensen's inequality.

The Gaussian Poincare inequality for \(\gamma\), together with the uniform
bounds on \(d\pi_n/d\gamma\), gives by Holley--Stroock
\[
      \Var_{\pi_n}(F)\le C\,{\cal E}_n(F,F)
\]
with \(C\) independent of \(n\).  Since \(P_t^n\) is the symmetric semigroup
generated by \({\cal E}_n\), the usual energy dissipation argument yields
\[
      \Var_{\pi_n}(P_t^n f)\le e^{-2t/C}\Var_{\pi_n}(f).
\]
Passing to the limit with (A) and (8), and then using density of \(C_b(E)\) in
\(L^2(\pi)\), proves (10).
\end{proof}

\section*{Fixed-time absolute continuity}

\begin{lemma}\label{lem:control}
For every deterministic \(x\in E\) and \(t>0\),
\[
        P_t(x,\cdot)\ll \gamma ,
\]
hence \(P_t(x,\cdot)\ll\pi\).
\end{lemma}

\begin{proof}
Write \(u_t=S_tx+V_t+K_t\).  Let \(r_k=t(1-2^{-k})\) and
\(\Delta_k=r_{k+1}-r_k\).  Set
\[
        D_k=K_{r_{k+1}}-S_{\Delta_k}K_{r_k}.
\]
By (B),
\[
        \E\|D_k\|_H^2\le C_t\,\Delta_k^{3/2},\qquad k\ge0.  \tag{11}
\]
Define the predictable \(H\)-valued process
\[
 F_s=\sum_{k\ge0}\frac{\1_{(r_{k+1},r_{k+2}]}(s)}{\Delta_{k+1}}\,
        S_{s-r_{k+1}}D_k .                                  \tag{12}
\]
Then
\[
 \E\int_0^t\|F_s\|_H^2ds
 \le C_t\sum_{k\ge0}\frac{\Delta_k^{3/2}}{\Delta_{k+1}}<\infty. \tag{13}
\]
Moreover the terminal effect of this adapted control is exactly \(K_t\):
\[
 \int_0^t S_{t-s}F_s\,ds
 =\lim_{N\to\infty}\sum_{k=0}^N S_{t-r_{k+1}}D_k
 =\lim_{N\to\infty}S_{t-r_{N+1}}K_{r_{N+1}}=K_t               \tag{14}
\]
in \(H\), by telescoping and continuity of \(K\).  Thus
\[
        u_t=S_tx+V_t+\int_0^tS_{t-s}F_s\,ds .                \tag{15}
\]

Let \(\tau_N=\inf\{s:\int_0^s\|F_r\|_H^2dr>N\}\) and
\(F^N=F\1_{[0,\tau_N]}\).  Novikov's condition is automatic for \(F^N\), and
the Hilbert-space Girsanov theorem, applied to the cylindrical Brownian motion
\(W\) in the given weak-solution filtration, implies that the terminal
\(H\)-law of
\[
 S_\cdot x+V_\cdot+\int_0^\cdot S_{\cdot-r}F^N_r\,dr
\]
is absolutely continuous with respect to the terminal Ornstein--Uhlenbeck law.
On \(\{\tau_N>t\}\) this stopped terminal value equals \(u_t\), while
\(\tau_N\uparrow\infty\) a.s.; hence \(\Law_H(u_t)\) is absolutely continuous
with respect to \(\Law_H(S_tx+V_t)\).

In the eigenbasis \(e_k(r)=\sqrt2\sin(k\pi r)\), the latter covariance has
eigenvalues
\[
        q_k(t)=\frac{1-e^{-k^2\pi^2t}}{k^2\pi^2},
\]
whereas \(\gamma\) has eigenvalues \((k^2\pi^2)^{-1}\).  The covariance ratios
differ from \(1\) by an exponentially square-summable sequence, and
\(S_tx\in H_0^1(0,1)\), the Cameron--Martin space of \(\gamma\).  The
Feldman--Hajek theorem gives \(\Law_H(S_tx+V_t)\sim\gamma\).  Since both
measures are supported on \(E\), and the Borel sets of \(E\) are traces of
\(H\)-Borel sets on the Borel subset \(E\subset H\), the same absolute
continuity holds for the \(E\)-valued laws.
\end{proof}

\section*{Conclusion: uniqueness}

Let \(\eta\) be any invariant probability measure.  If \(\pi(A)=0\), then
Lemma \ref{lem:control} gives \(P_t(y,A)=0\) for every \(y\in E\), so
\[
        \eta(A)=\int_E P_t(y,A)\,\eta(dy)=0.
\]
Thus \(\eta\ll\pi\); write \(d\eta=h\,d\pi\).

The detailed-balance identity extends from bounded continuous functions to all
bounded Borel functions because the finite measure
\(\pi(dx)P_t(x,dy)\) on \(E\times E\) is determined by \(C_b(E\times E)\).
Hence \(P_t\) is self-adjoint also in the \(L^1\)-\(L^\infty\) duality.
For bounded Borel \(f\),
\[
 \int f h\,d\pi=\int P_tf\,d\eta
      =\int P_tf\,h\,d\pi=\int f\,P_t h\,d\pi ,
\]
so \(P_t h=h\) in \(L^1(\pi)\).

For \(a>0\), let \(h_a=h\wedge a\).  Since \(r\mapsto r\wedge a\) is
concave,
\[
        P_t h_a\le (P_t h)\wedge a=h_a .
\]
The two sides have the same \(\pi\)-integral, hence \(P_t h_a=h_a\).  Now
\(h_a\in L^2(\pi)\), and the spectral gap (10) gives
\(\Var_\pi(h_a)=0\).  Therefore \(h_a\) is constant for every \(a\).  Letting
\(a\uparrow\infty\) and using \(\int h\,d\pi=1\), we obtain \(h\equiv1\).
Thus \(\eta=\pi\).

Consequently \(P_t\) has at most one invariant probability measure; in fact the
unique invariant measure is the Gibbs measure \(\pi\) in (7).
