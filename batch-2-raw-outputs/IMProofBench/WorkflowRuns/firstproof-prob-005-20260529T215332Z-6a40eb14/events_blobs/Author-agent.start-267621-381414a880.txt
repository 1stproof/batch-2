\begin{center}
{\bf Uniqueness of the invariant measure}
\end{center}

\section*{Problem statement and interpretation}

Let \(E=C_0([0,1])\), \(H=L^2(0,1)\), and let \(S_t\) be the Dirichlet heat
semigroup for \(\frac12\Delta\), with kernel \(p_t(x,y)\).  I use the
Dirichlet extension of the Athreya--Butkovsky--L\^e--Mytnik approximation
stipulated in the problem as follows.  For \(b_n=G_{1/n}\),
\[
  u_t^n=S_t x+V_t+K_t^n,\qquad
  K_t^n=\int_0^t S_{t-r}b_n(u_r^n)\,dr,                    \tag{1}
\]
where \(V\) is the Dirichlet stochastic convolution, and
\[
        P_t^n\Phi(x)\longrightarrow P_t\Phi(x),
        \qquad x\in E,\quad t>0,\quad \Phi\in C_b(E).        \tag{2}
\]
We shall not need a \(V(3/4)\)-estimate for the singular limiting drift.
Instead we prove a uniform Hilbert estimate for the smooth equations and pass
to the limit by lower semicontinuity of relative entropy.

\begin{lemma}[Uniform regularized Hilbert estimate]\label{lem:Hestimate}
For every \(T<\infty\) there is \(C_T\), independent of \(n\) and of the
initial point \(x\in E\), such that
\[
 \|K_t^n-S_{t-s}K_s^n\|_{L^2(\Omega;H)}
      \le C_T |t-s|^{3/4},\qquad 0\le s<t\le T .             \tag{3}
\]
\end{lemma}

\begin{proof}
Put \(D^n_{s,t}=K_t^n-S_{t-s}K_s^n\).  We first record the analytic estimates.
For \(d(y)=y\wedge(1-y)\) and
\[
        \rho_\tau(y)=\int_0^\tau p_{2a}(y,y)\,da ,
\]
standard Dirichlet heat-kernel bounds give, for \(0<\tau\le T\),
\[
        c(\sqrt{\tau}\wedge d(y))\le \rho_\tau(y)
        \le C(\sqrt{\tau}\wedge d(y)).                     \tag{4}
\]
Consequently
\[
\int_0^1\rho_\tau(y)^{-1/2}dy\le C\tau^{-1/4},\qquad
\sup_z\int_0^1 p_a(z,y)\rho_\tau(y)^{-1/2}dy
     \le C(a^{-1/4}+\tau^{-1/4}).                            \tag{5}
\]
Moreover \(\rho_\tau^{-1}\le C(\tau^{-1/2}+d^{-1})\).  Hardy's inequality on
\(H_0^1(0,1)\) and the heat-gradient estimate therefore imply
\[
        \|\rho_\tau^{-1}S_\tau f\|_H
        \le C\tau^{-1/2}\|f\|_H,\qquad f\in H .              \tag{6}
\]

For \(s<t\) define the frozen increment
\[
 A^n_{s,t}=\int_s^t S_{t-r}
 b_n\!\left(S_{r-s}u_s^n+V_r-S_{r-s}V_s\right)\,dr .          \tag{7}
\]
For fixed \(n\), \(D^n\) is the semigroup-sewn integral of \(A^n\): for any
partition \(\Pi\) of \([s,t]\),
\[
\sum_{[a,b]\in\Pi}S_{t-b}A^n_{a,b}
 \to \int_s^tS_{t-r}b_n(u_r^n)\,dr=D^n_{s,t}
\]
in \(L^2(H)\), because \(b_n\) is Lipschitz and
\(u_r^n-\{S_{r-a}u_a^n+V_r-S_{r-a}V_a\}=D^n_{a,r}=O_n(r-a)\) in \(H\).

First, uniformly in \(n\),
\[
        \|A^n_{s,t}\|_{L^2(\Omega;H)}\le C|t-s|^{3/4}.        \tag{8}
\]
Indeed, condition on \(\calF_s\), and write
\(Z_r=S_{r-s}u_s^n+V_r-S_{r-s}V_s\).  Since
\(Z_q(z)-S_{q-r}Z_r(z)\) is independent of \(\calF_r\) with variance
\(\rho_{q-r}(z)\), and \(b_n\) is a probability density,
\[
 \E_s\, b_n(Z_r(y))b_n(Z_q(z))
 \le C\,\rho_{r-s}(y)^{-1/2}\rho_{q-r}(z)^{-1/2}.
\]
Using \(\int p_a(x,y)p_b(x,z)\,dx=p_{a+b}(y,z)\) and (5),
\[
\begin{aligned}
 \E_s\|A^n_{s,t}\|_H^2
 &\le C\!\int_s^t\!\!\int_r^t\!\!
      \iint p_{2t-r-q}(y,z)
      \rho_{r-s}(y)^{-1/2}\rho_{q-r}(z)^{-1/2}\,dy\,dz\,dq\,dr  \\
 &\le C\!\int_s^t\!\!\int_r^t
      \bigl((2t-r-q)^{-1/4}+(r-s)^{-1/4}\bigr)(q-r)^{-1/4}
      \,dq\,dr
 \le C|t-s|^{3/2}.
\end{aligned}
\]

Second, if \(s<u<t\) and
\(\delta A^n_{s,u,t}=A^n_{s,t}-S_{t-u}A^n_{s,u}-A^n_{u,t}\), then
\[
        \|\E_u\delta A^n_{s,u,t}\|_H
        \le C(t-u)^{1/2}\|D^n_{s,u}\|_H .                   \tag{9}
\]
For \(r>u\), the two frozen arguments differ by \(h=S_{r-u}D^n_{s,u}\).
Conditional on \(\calF_u\), the remaining noise has variance
\(\rho_{r-u}(y)\), and hence
\[
 \left|\E_u\{b_n(Z_r(y))-b_n(Z_r(y)+h(y))\}\right|
      \le C\rho_{r-u}(y)^{-1}|h(y)|.
\]
Applying the \(L^2\)-contraction \(S_{t-r}\) and then (6) gives
\[
 \|S_{t-r}\E_u\{b_n(Z_r)-b_n(Z_r+h)\}\|_H
 \le C(r-u)^{-1/2}\|D^n_{s,u}\|_H,
\]
and integration over \(r\in[u,t]\) proves (9).

We use the following elementary Hilbert version of stochastic sewing.  If
\(S_t\) is a contraction semigroup and an adapted family \(A_{s,t}\) has
semigroup coboundary \(\delta A_{s,u,t}=A_{s,t}-S_{t-u}A_{s,u}-A_{u,t}\), then
the dyadic sewn limit \({\mathcal A}_{s,t}\), whenever it exists, satisfies
\[
\|{\mathcal A}_{s,t}-A_{s,t}\|_{L^2(H)}
 \le C\{\Gamma_1|t-s|^{1+\varepsilon_1}
        +\Gamma_2|t-s|^{1/2+\varepsilon_2}\},               \tag{10}
\]
provided, for dyadic midpoints \(u\),
\[
\|\E_s\delta A_{s,u,t}\|_{L^2(H)}\le\Gamma_1|t-s|^{1+\varepsilon_1},\quad
\|\delta A_{s,u,t}-\E_s\delta A_{s,u,t}\|_{L^2(H)}
     \le\Gamma_2|t-s|^{1/2+\varepsilon_2}.
\]
This is the \(L^2(H)\) case of L\^e's stochastic sewing lemma \cite{LeBanach,LeSSL}.  Directly,
compare two consecutive dyadic Riemann sums: the mean parts are summable by
the exponent \(1+\varepsilon_1>1\), while the centered parts are orthogonal
Hilbert-valued martingale differences, giving a square-sum of order
\(2^{-2\varepsilon_2 m}\).

Let
\[
Y_n(h)=\sup_{0<t-s\le h}
       \frac{\|D^n_{s,t}\|_{L^2(H)}}{|t-s|^{3/4}} ,
\]
which is finite for fixed \(n\) since \(b_n\) is bounded.  On intervals of
length at most \(h\), (9) gives
\[
 \|\E_a\delta A^n_{a,b,c}\|_{L^2(H)}
 \le C\,Y_n(h)(b-a)^{3/4}(c-b)^{1/2}
 \le C\,Y_n(h)(c-a)^{5/4}.
\]
Also (8) implies
\(\|\delta A^n_{a,b,c}-\E_a\delta A^n_{a,b,c}\|_{L^2(H)}
 \le C(c-a)^{3/4}\).  Applying (10) with
\(\varepsilon_1=\varepsilon_2=1/4\),
\[
 \|D^n_{s,t}-A^n_{s,t}\|_{L^2(H)}
 \le C|t-s|^{3/4}+C\,Y_n(h)|t-s|^{5/4}.
\]
Together with (8), this yields \(Y_n(h)\le C+C h^{1/2}Y_n(h)\).  Choosing
\(h>0\) small absorbs the last term, uniformly in \(n\).  Longer intervals are
split into subintervals of length at most \(h\) and are handled by
\(D^n_{s,t}=S_{t-r}D^n_{s,r}+D^n_{r,t}\).  This proves (3).
\end{proof}

\section*{The reversible Gibbs measure}

Let \(\gamma\) be Brownian bridge measure on \(E\), the centered Gaussian
measure with covariance \((-\Delta_D)^{-1}\).  Define
\[
  \Lambda(v)=\int_0^1\1_{\{v(r)>0\}}\,dr,\qquad
  \pi(dv)=Z^{-1}e^{2\Lambda(v)}\,\gamma(dv).                 \tag{11}
\]
For \(B_n'=b_n\), \(B_n(a)=\int_{-\infty}^a b_n(r)\,dr\), put
\[
   \pi_n(dv)=Z_n^{-1}\exp\left(2\int_0^1B_n(v(r))\,dr\right)\gamma(dv).
\]
Since \(0\le B_n\le1\), \(B_n(a)\to\1_{\{a>0\}}\) for \(a\ne0\), and a
Brownian bridge spends zero Lebesgue time at zero,
\[
        \|\pi_n-\pi\|_{\TV}\to0 .                            \tag{12}
\]
Moreover all densities \(d\pi_n/d\gamma\) and \(d\pi/d\gamma\) are bounded
above and below by deterministic positive constants.

For fixed \(n\), the smooth SPDE is the standard gradient diffusion associated
with
\[
        {\cal E}_n(F,G)=\frac12\int_E\langle DF,DG\rangle_H\,d\pi_n .
\]
On smooth cylinder functions,
\[
 L_nF(v)=\frac12{\rm Tr}\,D^2F(v)
       +\left\langle \frac12\Delta_Dv+b_n(v),DF(v)\right\rangle_H,
\]
and Gaussian integration by parts gives
\[
 \int_E G\,L_nF\,d\pi_n
 =-\frac12\int_E\langle DF,DG\rangle_H\,d\pi_n
 =\int_E F\,L_nG\,d\pi_n .                                  \tag{13}
\]
Thus \(P_t^n\) is \(\pi_n\)-invariant and reversible.

\begin{proposition}\label{prop:gap}
The measure \(\pi\) is \(P_t\)-invariant.  Moreover \(P_t\) is symmetric on
\(L^2(\pi)\), and there exists \(c>0\) such that
\[
       \Var_\pi(P_t f)\le e^{-2ct}\Var_\pi(f),
       \qquad f\in L^2(\pi),\ t\ge0 .                       \tag{14}
\]
\end{proposition}

\begin{proof}
For \(f,g\in C_b(E)\), (2), (12), and the invariance and symmetry of
\((P_t^n,\pi_n)\) imply
\[
 \int P_tf\,d\pi=\lim_n\int P_t^nf\,d\pi_n=\int f\,d\pi
\]
and
\[
 \int gP_tf\,d\pi=\lim_n\int gP_t^nf\,d\pi_n
       =\lim_n\int fP_t^ng\,d\pi_n=\int fP_tg\,d\pi .
\]
This extends to \(L^2(\pi)\) by density and Jensen's inequality.

The Gaussian Poincare inequality for \(\gamma\), together with the uniform
bounds on \(d\pi_n/d\gamma\), gives by Holley--Stroock
\[
      \Var_{\pi_n}(F)\le C\,{\cal E}_n(F,F)
\]
with \(C\) independent of \(n\).  Since \(P_t^n\) is the symmetric semigroup
generated by \({\cal E}_n\),
\[
      \Var_{\pi_n}(P_t^n f)\le e^{-2t/C}\Var_{\pi_n}(f).
\]
Passing to the limit with (2) and (12), then using density of \(C_b(E)\) in
\(L^2(\pi)\), proves (14).
\end{proof}

\section*{Fixed-time absolute continuity}

\begin{lemma}\label{lem:control}
For every deterministic \(x\in E\) and \(t>0\),
\[
        P_t(x,\cdot)\ll \gamma ,
\]
hence \(P_t(x,\cdot)\ll\pi\).
\end{lemma}

\begin{proof}
Let \(\nu_t^x=\Law(S_tx+V_t)\).  We first prove a uniform entropy bound for
the regularized terminal laws.  Let \(r_k=t(1-2^{-k})\) and
\(\Delta_k=r_{k+1}-r_k\).  For the \(n\)-th equation set
\(D_k^n=K^n_{r_{k+1}}-S_{\Delta_k}K^n_{r_k}\) and define the predictable
control
\[
 F_s^n=\sum_{k\ge0}\frac{\1_{(r_{k+1},r_{k+2}]}(s)}{\Delta_{k+1}}\,
        S_{s-r_{k+1}}D_k^n .                                \tag{15}
\]
Lemma \ref{lem:Hestimate} gives
\[
 \E\int_0^t\|F_s^n\|_H^2ds
 \le C_t\sum_{k\ge0}\frac{\Delta_k^{3/2}}{\Delta_{k+1}}<\infty, \tag{16}
\]
uniformly in \(n\).  The terminal effect telescopes:
\[
 \int_0^t S_{t-s}F_s^n\,ds
 =\lim_{N\to\infty}\sum_{k=0}^N S_{t-r_{k+1}}D_k^n
 =K_t^n                                                       \tag{17}
\]
in \(H\).  Hence
\(u_t^n=S_tx+V_t+\int_0^tS_{t-s}F_s^n\,ds\).

Let \(\tau_N=\inf\{s:\int_0^s\|F_r^n\|_H^2dr>N\}\) and
\(F^{n,N}=F^n\1_{[0,\tau_N]}\).  For the corresponding terminal variable
\[
 X_N=S_tx+V_t+\int_0^tS_{t-s}F_s^{n,N}ds ,
\]
Novikov's condition holds.  Hilbert-space Girsanov and the data-processing
inequality for relative entropy give
\[
 \Ent(\Law(X_N)\mid\nu_t^x)
 \le \frac12\,\E\int_0^t\|F_s^{n,N}\|_H^2ds
 \le \frac12\,\E\int_0^t\|F_s^n\|_H^2ds .
\]
On the event \(\{\tau_N>t\}\), \(X_N=u_t^n\), and this event has probability
tending to one by (16).  Hence \(X_N\Rightarrow u_t^n\) (as \(E\)-valued
variables; equivalently one may view the supported laws on \(H\)).  Lower
semicontinuity of entropy and (16) yield
\[
        \sup_n \Ent(P_t^n(x,\cdot)\mid \nu_t^x)<\infty .      \tag{18}
\]
By (2), \(P_t^n(x,\cdot)\Rightarrow P_t(x,\cdot)\) on the Polish space \(E\);
relative entropy is lower semicontinuous under weak convergence, so
\[
        \Ent(P_t(x,\cdot)\mid\nu_t^x)<\infty .
\]
Thus \(P_t(x,\cdot)\ll\nu_t^x\).

Finally, in the Dirichlet eigenbasis \(e_k(r)=\sqrt2\sin(k\pi r)\), the
covariance of \(\nu_t^x\) has eigenvalues
\[
        q_k(t)=\frac{1-e^{-k^2\pi^2t}}{k^2\pi^2},
\]
while \(\gamma\) has eigenvalues \((k^2\pi^2)^{-1}\).  The covariance ratios
differ from \(1\) by an exponentially square-summable sequence, and
\(S_tx\in H_0^1(0,1)\), the Cameron--Martin space of \(\gamma\).  The
Feldman--Hajek theorem gives \(\nu_t^x\sim\gamma\).  Therefore
\(P_t(x,\cdot)\ll\gamma\sim\pi\).
\end{proof}

\section*{Conclusion: uniqueness}

Let \(\eta\) be any invariant probability measure.  If \(\pi(A)=0\), then
Lemma \ref{lem:control} gives \(P_t(y,A)=0\) for every \(y\in E\), so
\[
        \eta(A)=\int_E P_t(y,A)\,\eta(dy)=0.
\]
Thus \(\eta\ll\pi\); write \(d\eta=h\,d\pi\).

The detailed-balance identity extends from bounded continuous functions to all
bounded Borel functions by a monotone-class argument applied to the finite
measure \(\pi(dx)P_t(x,dy)\) on \(E\times E\).  Hence \(P_t\) is self-adjoint
also in \(L^1\)-\(L^\infty\) duality.  For bounded Borel \(f\),
\[
 \int f h\,d\pi=\int P_tf\,d\eta
      =\int P_tf\,h\,d\pi=\int f\,P_t h\,d\pi ,
\]
so \(P_t h=h\) in \(L^1(\pi)\).

For \(a>0\), let \(h_a=h\wedge a\).  Since \(r\mapsto r\wedge a\) is concave,
\[
        P_t h_a\le (P_t h)\wedge a=h_a .
\]
The two sides have the same \(\pi\)-integral, hence \(P_t h_a=h_a\).  Now
\(h_a\in L^2(\pi)\), and the spectral gap (14) gives
\(\Var_\pi(h_a)=0\).  Therefore \(h_a\) is constant for every \(a\).  Letting
\(a\uparrow\infty\) and using \(\int h\,d\pi=1\), we obtain \(h\equiv1\).
Thus \(\eta=\pi\).

Consequently \(P_t\) has at most one invariant probability measure; in fact the
unique invariant measure is the Gibbs measure \(\pi\) in (11).
