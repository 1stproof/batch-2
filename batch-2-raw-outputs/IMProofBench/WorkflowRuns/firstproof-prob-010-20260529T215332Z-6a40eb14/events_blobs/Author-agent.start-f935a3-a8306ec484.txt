\title{Proper proximality of irreducible graph-product von Neumann algebras}
\author{}

\section*{Problem statement and interpretation}

I interpret irreducibility in the usual graph-product sense: the complement
graph $\Gamma^c$ is connected, equivalently $\Gamma$ is not a non-trivial join.
The traces are faithful normal tracial states.  For $U\subset V$ let $M_U$
denote the graph product over the induced subgraph on $U$, and let
$E_U:M\to M_U$ be the canonical trace-preserving expectation.

We use the following notation from Ding--Kunnawalkam Elayavalli--Peterson
\cite{DKEP}.  If $B\subset N$ is an inclusion with expectation, $X_B$ denotes
the boundary piece generated by the Jones projection $e_B:L^2(N)\to L^2(B)$,
and $S_{X_B}(N)$ is the associated small-at-infinity operator system.  Proper
proximality relative to $X_B$ means that there is no $N$-central state on
$S_{X_B}(N)$ whose restriction to $N$ is normal.

\section*{Boundary-piece facts}

We shall use two standard consequences of the boundary-piece formalism of
\cite[Sections 3 and 6]{DKEP}.  Write $q_X$ for the support projection of a
boundary piece $X$ in the normal bidual picture of \cite[Lemma 8.5]{DKEP}, and
write $q_{\cK}$ for the support of the compact boundary piece.

\begin{lemma}[finite boundary criterion]\label{lem:finite-boundary}
Let $N$ be a separable non-amenable finite factor.  Let $X_1,\ldots,X_m$ be
boundary pieces for $N$ such that
\[
        q_{X_1}\wedge\cdots\wedge q_{X_m}=q_{\cK}.
\]
If $N$ is properly proximal relative to each $X_i$, then $N$ is properly
proximal.
\end{lemma}

\begin{proof}
We argue by contradiction using the normal-bidual characterization of proper
proximality \cite[Lemma 8.5]{DKEP}.  If $N$ is not properly proximal, there is
an $N$-central state $\omega$ on the bidual small-at-infinity system
$\widetilde S(N)$ with $\omega|_N=\tau$.

Fix $i$.  The boundary calculus gives
\[
        q_{X_i}^{\perp}\widetilde S_{X_i}(N)q_{X_i}^{\perp}
        \subset \widetilde S(N).
\]
If $\omega(q_{X_i}^{\perp})>0$, then
$T\mapsto \omega(q_{X_i}^{\perp}Tq_{X_i}^{\perp})/\omega(q_{X_i}^{\perp})$
is an $N$-central state on $\widetilde S_{X_i}(N)$ normal on $N$, contradicting
proper proximality relative to $X_i$.  Hence $\omega(q_{X_i}^{\perp})=0$ for
all $i$, so $\omega(q_{\cK})=1$.

Thus $\omega$ is supported on the compact boundary.  The compact corner in the
normal-bidual picture is the normal bidual of $\mathbb B(L^2N)$, so the
restriction gives an $N$-central state on $\mathbb B(L^2N)$ extending $\tau$.
By Connes's amenability criterion this makes the finite factor $N$ amenable,
contrary to the hypothesis.
\end{proof}

For a graph product $M=M_V$ and $U\subset V$, let $X_U$ denote the boundary
piece $X_{M_U}$.

\begin{lemma}[the link boundary pieces fill the boundary]\label{lem:links-fill}
For every finite graph product,
\[
        \bigwedge_{v\in V} q_{X_{\Lk(v)}}=q_{\cK}.
\]
\end{lemma}

\begin{proof}
The boundary piece $X_U$ is the one associated to the correspondence
$L^2(M)\otimes_{M_U}L^2(M)$.  The fusion rule of
Charlesworth--de Santiago--Hayes--Jekel--Kunnawalkam Elayavalli--Nelson
\cite[Main Theorem 0.4]{CDHJKN} implies that the product of the support
projections attached to $U_1,\ldots,U_r$ is the support projection attached to
$U_1\cap\cdots\cap U_r$.  Indeed, their formula decomposes the relative tensor
product of the corresponding bimodules as a direct sum of the bimodules
attached to subgraphs contained in the intersection, and hence no larger
boundary piece survives.

Applying this with $U_v=\Lk(v)$ gives the compact boundary, because
$\bigcap_{v\in V}\Lk(v)=\varnothing$ (there are no loops, so $v\notin\Lk(v)$)
and $M_{\varnothing}=\mathbb C$; the boundary piece generated by
$e_{\mathbb C}$ is precisely $\cK(L^2M)$.
\end{proof}

\section*{A relative Bass--Serre paradox}

\begin{proposition}\label{prop:relative-BS}
Let
\[
        N=P*_{B}Q,\qquad Q=B\bbarotimes A,
\]
with trace-preserving expectations.  Suppose that $A$ contains a trace-zero
unitary $a$, and that $P$ contains unitaries $w_1,w_2$ satisfying
\[
        E_B(w_i)=0,\qquad E_B(w_1^*w_2)=0.
\]
Then $N$ is properly proximal relative to the boundary piece $X_B$.
\end{proposition}

\begin{proof}
Write the amalgamated free-product Fock decomposition over $B$ as
\[
L^2(N)=L^2(B)\oplus
\bigoplus H_{i_1}^\circ\otimes_B\cdots\otimes_B H_{i_n}^\circ,
\]
where $i_j\in\{P,Q\}$ alternate and
$H_P^\circ=L^2(P)\ominus L^2(B)$, $H_Q^\circ=L^2(Q)\ominus L^2(B)$.
Let $p$ and $q$ be the projections onto the sums of reduced words whose first
letter is respectively in $H_P^\circ$ and $H_Q^\circ$.  Then
$1=e_B+p+q$.

For $x$ in the algebraic free product, right multiplication by $x$ can change
the first letter only through cancellations landing in $B$; the corresponding
operators factor through $L^2(B)$.  Thus $[p,JxJ]$ and $[q,JxJ]$ lie in
$K^{\infty,1}_{X_B}(N)$.  By \cite[Lemma 6.1]{DKEP}, $p,q\in S_{X_B}(N)$.
Also $e_B\in X_B\subset S_{X_B}(N)$.

Let $\varphi$ be an $N$-central state on $S_{X_B}(N)$ normal on $N$.  Since
$E_B(w_i)=0$, left multiplication by $w_i$ maps the $Q$-initial summand into
the $P$-initial summand.  The two images are orthogonal because
$E_B(w_1^*w_2)=0$.  Hence
\[
        w_1qw_1^*+w_2qw_2^*\le p .
\]
Similarly $a\in Q\ominus B$ gives $apa^*\le q$.  Centrality yields
\[
        2\varphi(q)\le \varphi(p)\le \varphi(q),
\]
and therefore $\varphi(p)=\varphi(q)=0$.

Finally, $h=w_1a$ is a $B$-Haar unitary: the subspaces
$h^nL^2(B)$, $n\in\mathbb Z$, are pairwise orthogonal, since $h^n$ is a reduced
alternating word with $B$-expectation zero for $n\ne0$.  Hence the projections
$h^ne_Bh^{-n}$ are pairwise orthogonal.  As $S_{X_B}(N)$ is an $N$-bimodule,
these projections belong to $S_{X_B}(N)$, and centrality gives
$m\varphi(e_B)\le1$ for every $m\in\mathbb N$.  Thus $\varphi(e_B)=0$, which
contradicts $1=e_B+p+q$.
\end{proof}

\section*{Application to the graph product}

For each vertex $v$ put $L_v=\Lk_\Gamma(v)$ and
$P_v=M_{V\setminus\{v\}}$.  The standard graph-product decomposition is
\[
        M\cong P_v*_{M_{L_v}}\bigl(M_{L_v}\bbarotimes M_v\bigr).
\]
We first check that Proposition \ref{prop:relative-BS} applies to this
decomposition for every $v$.

Fix $v$.  Let
$C(v)=\{r\ne v: r\not\sim_\Gamma v\}$ be the set of complement-neighbours of
$v$.  Since $\Gamma^c$ is connected and $|V|\ge3$, $C(v)\ne\varnothing$.
If $C(v)$ contains two distinct vertices $r,t$, choose trace-zero unitaries
$u_r\in M_r$ and $u_t\in M_t$ and put $w_1=u_r$, $w_2=u_t$.  The expectation
onto $M_{L_v}$ kills $w_1,w_2,w_1^*w_2$, because each word contains a centered
letter from a vertex outside $L_v$.

If $C(v)=\{r\}$, then connectedness of $\Gamma^c$ gives a vertex
$s\ne v,r$ with $r\not\sim_\Gamma s$; necessarily $s\in L_v$.  Choose
trace-zero unitaries $u_r\in M_r$, $u_s\in M_s$, and set
$w_1=u_r$, $w_2=u_su_r$.  Again
\[
        E_{M_{L_v}}(w_i)=0,\qquad
        E_{M_{L_v}}(w_1^*w_2)=E_{M_{L_v}}(u_r^*u_su_r)=0,
\]
because the displayed words are reduced and contain the outside vertex $r$.
Thus $M$ is properly proximal relative to $X_{L_v}$ for every $v$.

It remains to note that $M$ is a non-amenable factor.  By
\cite[Main Theorem 0.5]{CDHJKN}, factoriality can fail only if there is a
vertex adjacent to all other vertices, or a non-adjacent pair adjacent to all
remaining vertices.  Either situation would give an isolated vertex, respectively
an isolated edge, in the connected complement graph $\Gamma^c$, impossible since
$|V|\ge3$.  Thus $M$ is a factor.  By \cite[Main Theorem 0.1]{CDHJKN}, if $M$
were amenable, then every non-edge $r\not\sim s$ would have to be one of the
two-dimensional exceptional pairs adjacent to all other vertices.  Such a pair
would again be an isolated component of $\Gamma^c$, contradicting connectedness
and $|V|\ge3$.  Hence $M$ is non-amenable.

The boundary pieces $X_{L_v}$ fill the boundary by Lemma \ref{lem:links-fill}.
Since $M$ is properly proximal relative to each of them, Lemma
\ref{lem:finite-boundary} gives that $M$ is properly proximal.
